if(!nameInput) return alert("Please enter a name"); currentUser = nameInput; // Check if user exists, if not, create them in the cloud db.ref('users/' + currentUser).once('value', (snapshot) => { if (!snapshot.exists()) { // CREATE NEW ACCOUNT db.ref('users/' + currentUser).set({ username: currentUser, bio: "New Artist", joined: Date.now() }).then(() => { console.log("New account created!"); proceedToApp(); }); } else { // LOG IN TO EXISTING console.log("Welcome back!"); proceedToApp(); } }); } function proceedToApp() { document.getElementById('auth').style.display = 'none'; document.getElementById('my-name').innerText = currentUser; // Load the public data syncGallery(); syncGroups(); // Load personal profile data db.ref('users/' + currentUser + '/pfp').on('value', (snap) => { if(snap.val()) { document.getElementById('my-pfp').style.backgroundImage = `url(${snap.val()})`; } }); } // --- PUBLIC GALLERY LOGIC --- function uploadArt(input) { const file = input.files[0]; if(!file) return; const reader = new FileReader(); reader.onload = (e) => { const img = new Image(); img.src = e.target.result; img.onload = () => { const canvas = document.createElement('canvas'); const max = 800; let w = img.width, h = img.height; if(w > h) { if(w > max){ h *= max/w; w = max; } } else { if(h > max){ w *= max/h; h = max; } } canvas.width = w; canvas.height = h; canvas.getContext('2d').drawImage(img, 0, 0, w, h); const data = canvas.toDataURL('image/jpeg', 0.6); db.ref('community_gallery').push({ image: data, author: currentUser, timestamp: Date.now() }); }; }; reader.readAsDataURL(file); } function syncGallery() { db.ref('community_gallery').on('value', (snap) => { const data = snap.val(); const globalFeed = document.getElementById('global-gallery'); const personalFeed = document.getElementById('my-gallery'); globalFeed.innerHTML = personalFeed.innerHTML = ""; if(data) { const items = Object.values(data).reverse(); items.forEach(item => { const html = `
@${item.author}
`; globalFeed.innerHTML += html; if(item.author === currentUser) personalFeed.innerHTML += html; }); } }); } function syncGroups() { db.ref('groups').on('value', (snap) => { const data = snap.val(); const groupList = document.getElementById('group-list'); groupList.innerHTML = ""; if(data) { Object.values(data).forEach(g => { groupList.innerHTML += `

${g.name}

Code: ${g.code}

`; }); } }); } function createGroup() { const name = prompt("Studio Name:"); if(name) { db.ref('groups').push({ name, code: Math.floor(1000 + Math.random() * 9000), creator: currentUser }); } } function show(id, el) { document.querySelectorAll('.page').forEach(p => p.classList.remove('active-page')); document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active')); document.getElementById(id).classList.add('active-page'); el.classList.add('active'); }
